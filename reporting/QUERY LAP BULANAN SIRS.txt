 KT.NAMA_KOTA, DIAGNOSA_ID, PD.GENDER, AGEYEAR, AGEMONTH, AGEDAY, 
 SUFFER_TYPE, PD.STATUS_PASIEN_ID, CLINIC_ID, KELUAR_ID
FROM         PASIEN_DIAGNOSA PD LEFT OUTER JOIN PASIEN P  ON PD.NO_REGISTRATION = P.NO_REGISTRATION
LEFT OUTER JOIN KALURAHAN K ON P.KAL_ID = K.KAL_ID
LEFT OUTER JOIN KECAMATAN KC ON KC.KEC_ID = K.KEC_ID
LEFT OUTER JOIN KOTA KT ON KC.KODE_KOTA = KT.KODE_KOTA

AND (CLINIC_ID > 'A010') AND (CLINIC_ID < 'O000')

       PASIEN.KK_NO,
       PASIEN.ORG_UNIT_CODE,

         PASIEN.death_date,
         PASIEN.GENDER,
